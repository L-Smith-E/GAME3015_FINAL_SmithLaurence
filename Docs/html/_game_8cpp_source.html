465void Game::BuildRootSignature()
466{
467 CD3DX12_DESCRIPTOR_RANGE texTable;
468 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
469
470 // Root parameter can be a table, root descriptor or root constants.
471 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
472
473 // Perfomance TIP: Order from most frequent to least frequent.
474 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
475 slotRootParameter[1].InitAsConstantBufferView(0);
476 slotRootParameter[2].InitAsConstantBufferView(1);
477 slotRootParameter[3].InitAsConstantBufferView(2);
478
479 auto staticSamplers = GetStaticSamplers();
480
481 // A root signature is an array of root parameters.
482 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
483 //define an array of so - called static samplers your application can use.
484 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
485 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
486 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
487
488 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
489 ComPtr<ID3DBlob> serializedRootSig = nullptr;
490 ComPtr<ID3DBlob> errorBlob = nullptr;
491 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
492 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
493
494 if (errorBlob != nullptr)
495 {
496 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
497 }
498 ThrowIfFailed(hr);
499
500 ThrowIfFailed(md3dDevice->CreateRootSignature(
501 0,
502 serializedRootSig->GetBufferPointer(),
503 serializedRootSig->GetBufferSize(),
504 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
505}

798{

814 //step18

816 tex.Offset(ri->Mat->DiffuseSrvHeapIndex, mCbvSrvDescriptorSize);

821 //step19
822 cmdList->SetGraphicsRootDescriptorTable(0, tex);
823 cmdList->SetGraphicsRootConstantBufferView(1, objCBAddress);
824 cmdList->SetGraphicsRootConstantBufferView(3, matCBAddress);
